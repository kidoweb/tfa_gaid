Полный гайд по созданию оружия в TFA

Основные настройки

Переменная Описание
SWE.Base Базовый класс оружия, от которого наследуется это оружие
SWE.Category Категория, в которой будет отображаться оружие в меню спавна
SWE.Spawnable Можно ли обычным игрокам брать это оружие (true/false)
SWE.AdminSpawnable Можно ли администраторам брать это оружие (true/false)

Параметры боеприпасов

Переменная Описание
SWE.Primary.ClipSize Размер магазина (сколько патронов помещается)
SWE.Primary.DefaultClip Начальное количество патронов (в запасе + в магазине)
SWE.Primary.Ammo Тип боеприпасов (например, "pistol", "rifle", "shotgun")

Характеристики стрельбы

Переменная Описание
SWE.Primary.RPM Скорострельность (выстрелы в минуту)
SWE.Primary.Damage Урон одного выстрела
SWE.Primary.NumShots Количество пуль за один выстрел (для дробовиков может быть больше 1)
SWE.Primary.Spread Разброс пуль при стрельбе без прицеливания
SWE.Primary.IronAccuracy Точность при стрельбе с прицеливанием

Настройки прицеливания

Переменная Описание
SWE.data.ironsights Включение системы прицеливания (1 - включено, 0 - выключено)
SWE.Secondary.IronFOV Поле зрения (Field of View) при прицеливании

Пример полного кода оружия


SWE.Base = "tfa_gun_base"
SWE.Category = "Custom Weapons"
SWE.Spawnable = true
SWE.AdminSpawnable = true
// Параметры боеприпасов
SWE.Primary.ClipSize = 30
SWE.Primary.DefaultClip = 120
SWE.Primary.Ammo = "pistol"
// Характеристики стрельбы
SWE.Primary.RPM = 600
SWE.Primary.Damage = 25
SWE.Primary.NumShots = 1
SWE.Primary.Spread = 0.02
SWE.Primary.IronAccuracy = 0.01
// Прицеливание
SWE.data = {}
SWE.data.ironsights = 1
SWE.Secondary.IronFOV = 75
// Модели
SWE.ViewModel = "models/weapons/v_pistol.mdl"
SWE.WorldModel = "models/weapons/w_pistol.mdl"
// Звуки
SWE.Primary.Sound = Sound("Weapon_Pistol.Single")
SWE.Primary.ReloadSound = Sound("Weapon_Pistol.Reload")
// Анимации
SWE.Animations = {
    ["reload"] = {
        ["type"] = TFA.Enum.ANIMATION_ACT, 
        ["value"] = ACT_VM_RELOAD
    }
}
// Другие настройки
SWE.HoldType = "pistol"
SWE.Slot = 1
SWE.SlotPos = 2
SWE.AutoSwitchTo = false
SWE.AutoSwitchFrom = false
